PCA

代码部分

# scaling variables
from sklearn import preprocessing
for col in df.columns:
    df[col] = preprocessing.scale(df[col])

# Fit a PCA model
from sklearn.preprocessing import Imputer
from sklearn.decomposition import PCA
p = 10                                     # 主成分数量
model = PCA(n_components=p)
model.fit(df)
lambda = model.explained_variance_ratio_   # 贡献率
print("Explained variance ratio of principal components:\n",lambda) 

# Report PVE and CVE plots
cve = np.cumsum(lambda)                     # 累积贡献率
plt.subplots(1,2,figsize=(10,4))   
plt.subplot(1,2,1)
plt.plot(range(1,p+1),lambda,'k-',linewidth=1,marker='.',markersize=8)
plt.plot([1,p],[0.95,0.95],'r--')          # 目标95%
plt.ylim(-0.1,1.1)
plt.xlabel('Principal Component')
plt.ylabel('Prop. Variance Explained')
plt.title("PVE plots")

plt.subplot(1,2,2)
plt.plot(range(1,p+1),cve,'k-',linewidth=1,marker='.',markersize=8)
plt.ylim(0.5,1.1)
plt.plot([1,p],[0.95,0.95],'r--')
plt.xlabel('Principal Component')
plt.ylabel('Cumulative Prop. Variance Explained')
plt.title("CVE plots")

plt.tight_layout()
plt.show()

# Calculate eigenvalues by hand
import numpy.linalg as la
covx = np.cov(df.values.T)      # covariance matrix
u,v = la.eig(covx)              # u is eigenvalue, v is direction
print('Eigenvalues calculated locally:\n',np.round(u[:p],2))
print('Eigenvalues calculated by sklearn.decomposition.PCA:\n',np.round(model.explained_variance_,2))
print('CVE calculated locally:\n',np.round(u[:p]/sum(u),2))
print('CVE calculated by sklearn.decomposition.PCA:\n',np.round(model.explained_variance_ratio_,2))
