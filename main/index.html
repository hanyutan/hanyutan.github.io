Monty Hall问题

from random import choice
def monty_hall(doors_num=3,decision='change',experiment=1000):
    """在给定选择下，模拟n次Monty Hall实验"""
    doors = list(range(1,doors_num+1))      # 门的序号
    results = []                            # 储存每次结果的列表
    
    for i in range(experiment): 
        succ = choice(doors)                # 设置中奖门号
        fail = doors.copy()
        fail.remove(succ)                   # 未中奖门号列表
        choose =  choice(doors)             # 挑战者随机挑选一扇门
        unchoose = doors.copy()
        unchoose.remove(choose)             # 未被挑选的门号列表
        open1 = choice([x for x in fail if x in unchoose])   # 主持人开一扇未中奖的门
        unchoose.remove(open1)
        if decision == 'change':            # 是否更改选择
            choose = unchoose[0]
        results.append(succ == choose)      # 记录是否中奖

    succ_rate = sum(results)/experiment     # 计算中奖概率
    return succ_rate
	
# 更改选择后的中奖概率2/3
monty_hall(decision='change',experiment=10000)
# 不换选择后的中奖概率1/3
monty_hall(decision='not change',experiment=10000)
